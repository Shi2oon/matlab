clear;
t = linspace(0,2*pi,500)';
y =  100*sin(t);
noise = normrnd(0,15,500,1);%噪声产生，均值0，标准差15，500个数值
y = y + noise;
figure;
plot(t,y);
xlabel('t')
ylabel('y = sin(t) + 噪声');
